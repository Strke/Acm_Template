###  卢卡斯定理（求组合数）

```c++
#include<iostream>
using namespace std;
typedef long long ll;
int p;
int qmi(int a,int k){
    int res=1;
    while(k){
        if(k&1) res=(ll)res*a%p;
        a=(ll)a*a%p;
        k>>=1;
    }
    return res;
}
int C(ll a,ll b){
    if(a<b) return 0;
    int res=1;
    for(int i=1,j=a;i<=b;i++,j--){
        res=(ll)res*j%p;
        res=(ll)res*qmi(i,p-2)%p;
    }
    return res;
} 
int lucas(ll a,ll b){
    if(a<p && b<p) return C(a,b);
    return (ll)C(a%p,b%p)*lucas(a/p,b/p)%p;
}
int main(){
    int n;cin>>n;
    ll a,b;
    while(n--){
        cin>>a>>b>>p;
        cout<<lucas(a,b)<<endl;
    }
    return 0;
}
```
