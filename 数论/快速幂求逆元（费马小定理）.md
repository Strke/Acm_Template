###  快速幂求逆元（费马小定理）



```c++
#include<iostream>
using namespace std;
typedef long long ll;
ll qmi(ll x,ll y,ll p){
    ll res=1;
    while(y){
        if(y&1) res=(res*x)%p;
        x=(x*x)%p;
        y=y>>1;
    }
    return res;
}
int main(){
    int n;cin>>n;
    while(n--){
        ll a,p;cin>>a>>p;
        ll res=qmi(a,p-2,p);
        if(a%p) cout<<res<<endl;
        else cout<<"impossible"<<endl;
    }
    return 0;
}
```
