### 扩展欧几里得

```c++
#include<iostream>
using namespace std;
int exgcd(int a,int b,int &x,int &y){
    if(b==0){
        x=1;y=0;
        return a;
    }
    int d=exgcd(b, a%b, x, y );
    int t=y;
    y=x-a/b*y;
    x=t;
    return d;
}
int main(){
    int n;cin>>n;
    while(n--){
        int a,b;cin>>a>>b;
        int x,y;
        exgcd(a,b,x,y);
        cout<<x<<" "<<y<<endl;
    }
    return 0;
}
```
